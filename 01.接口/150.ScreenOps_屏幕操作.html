ScreenOps_屏幕操作 ​

模块简介 ​

ScreenOps 提供屏幕信息获取、屏幕方向/全屏控制，以及全屏 Bitmap 截图能力，需先初始化上下文以避免获取屏幕信息时异常。

⚠️ 不支持懒人精灵最高权限下运行(最高权限下获取不到Context)。

全屏截图说明 ​

  • 不依赖 MediaProjection，需设备 root。
  • snapShotByShell：系统 screencap。
  • snapShotByScrcpy：移植 scrcpy 特权截图；
  • snapShot：优先 scrcpy，失败回退 screencap。
  • screenShot(Activity) 只截当前 Activity 窗口，不能替代全屏截图。

功能概览 ​

  • 上下文设置：设置 Context（setContext，与 UtilsMain.init 二选一）
  • 尺寸与密度：应用区域宽高、物理宽高、密度与 DPI
  • 方向与锁屏：横竖屏判断、锁屏状态查询
  • 亮灭与休眠：休眠时长设置/获取
  • 全屏与方向控制：设置全屏、横屏、竖屏（需 Activity）
  • 全屏截图：root screencap / 内置 scrcpy-server
  • 自动亮度：判断是否开启自动亮度
  • 亮度获取/设置：读取与设置系统亮度（0-255）
  • 高度获取：状态栏、ActionBar、导航栏高度
  • 支持性判断：是否支持导航栏

API 接口列表 ​

1. 初始化 ​

setContext() ​

按需设置 Context（建议传入 applicationContext），在未调用 UtilsMain.init 时也可提前注入。

setContext()和 UtilsMain.init() 二选一调用即可。

参数名类型必填说明
contextContext是应用或 Activity 的 context，建议 applicationContext
返回值类型说明
void无

示例：

lua
import('com.nx.assist.lua.LuaEngine')

local scOps = UtilsMain.screenOps()

local context = LuaEngine.getContext()
scOps.setContext(context)  -- 传入已有的 context

5. 全屏截图（需 Root） ​

isScrcpyReady() ​

内置 scrcpy-server.jar 是否可读。

返回值类型说明
Boolean就绪返回 true

snapShot() ​

全屏/区域截图。左右上下均为 0 时截整屏。优先 scrcpy，失败回退 root screencap。

参数名类型必填说明
leftInt是左边界
topInt是上边界
rightInt是右边界
bottomInt是下边界
返回值类型说明
Bitmap成功返回 Bitmap，失败返回 null

示例：

lua
local UtilsMain = loader.loadClass('com.xfx.plugin.UtilsMain')
UtilsMain.init(LuaEngine.getContext())
local scOps = UtilsMain.screenOps()

local bmp = scOps.snapShot(0, 0, 0, 0)
if bmp ~= nil then
  print('w=', bmp.getWidth(), 'h=', bmp.getHeight())
  -- 用完后按需 bmp.recycle()
end

OOP调用例子 ​

lua
-- ============================================
-- 屏幕尺寸获取示例
-- 演示如何使用 XfxPlugin.lua 调用 ScreenOps 获取屏幕信息
-- ============================================

-- 使用 XfxPlugin.lua 面向对象封装（推荐）
-- ============================================
print('========== 使用 XfxPlugin.lua 封装 ==========')

import('com.nx.assist.lua.LuaEngine')

-- 加载 XfxPlugin.lua 类
local xfxModule = require('lib/XfxPlugin')

-- 创建 XFX 对象实例
local XFX = xfxModule:new({
    apkName = 'xfxPlugin-release.apk',  -- APK 文件名/路径
    context = LuaEngine.getContext(),  -- 上下文对象（不支持最高权限）
})

-- 方法 1：通过 call 方法调用 ScreenOps 的方法
print('\n--- 通过 call 方法调用 ---')
-- XFX:call('screenOps', 'setContext', LuaEngine.getContext())

local ok1, screenWidth = pcall(function()
    return XFX:call('screenOps', 'getScreenWidth')
end)
local ok2, screenHeight = pcall(function()
    return XFX:call('screenOps', 'getScreenHeight')
end)

if ok1 and ok2 then
    print('屏幕宽度: ' .. tostring(screenWidth))
    print('屏幕高度: ' .. tostring(screenHeight))
else
    print('获取屏幕尺寸失败: ' .. tostring(screenWidth or screenHeight))
end

-- 方法 2：直接获取 screenOps 对象，然后调用方法（更直观）
print('\n--- 直接获取 screenOps 对象 ---')
local scOps = XFX:getOps('screenOps')
-- scOps.setContext(LuaEngine.getContext())

if scOps then
    -- 使用 pcall 捕获可能的初始化错误
    local ok, width = pcall(function() return scOps.getScreenWidth() end)
    local ok2, height = pcall(function() return scOps.getScreenHeight() end)
    
    if not ok or not ok2 then
        print('错误: ' .. tostring(width or height))
        print('提示: ScreenOps 需要先初始化 UtilsMain.init(context)')
        print('注意: 如果使用 XfxPlugin.lua，new 方法会自动初始化')
        -- 不 return，继续执行其他示例
    else
        -- 成功获取尺寸，继续获取其他信息
        local appWidth = scOps.getAppScreenWidth()  -- 应用屏幕宽度（排除状态栏等）
        local appHeight = scOps.getAppScreenHeight()  -- 应用屏幕高度
        local density = scOps.getDensity()  -- 屏幕密度
        local densityDpi = scOps.getDensityDpi()  -- 屏幕 DPI
        local xDpi = scOps.getXDpi()  -- X 方向 DPI
        local yDpi = scOps.getYDpi()  -- Y 方向 DPI
        local isLandscape = scOps.isLandscape()  -- 是否横屏
        local isPortrait = scOps.isPortrait()  -- 是否竖屏
        
        print('屏幕宽度: ' .. tostring(width))
        print('屏幕高度: ' .. tostring(height))
        print('应用屏幕宽度: ' .. tostring(appWidth))
        print('应用屏幕高度: ' .. tostring(appHeight))
        print('屏幕密度: ' .. tostring(density))
        print('屏幕 DPI: ' .. tostring(densityDpi))
        print('X 方向 DPI: ' .. tostring(xDpi))
        print('Y 方向 DPI: ' .. tostring(yDpi))
        print('是否横屏: ' .. tostring(isLandscape))
        print('是否竖屏: ' .. tostring(isPortrait))
    end
end

-- 方法 3：结合悬浮日志窗口显示屏幕信息（更友好）
print('\n--- 使用悬浮日志窗口显示 ---')
-- 确保有屏幕尺寸数据
if screenWidth and screenHeight then
    XFX:showFloatLogWindow({
        title = '屏幕信息',
        width = math.floor(screenWidth / 3),
        height = math.floor(screenHeight / 6),
    })
    XFX:setFloatLogTitle('屏幕尺寸信息')
    XFX:logi('宽度: ' .. tostring(screenWidth))
    XFX:logi('高度: ' .. tostring(screenHeight))
    if scOps then
        local ok, dpi = pcall(function() return scOps.getDensityDpi() end)
        local ok2, density = pcall(function() return scOps.getDensity() end)
        if ok then XFX:logi('DPI: ' .. tostring(dpi)) end
        if ok2 then XFX:logi('密度: ' .. tostring(density)) end
    end
else
    print('无法显示悬浮窗：屏幕尺寸获取失败')
end

sleep(5000) 
XFX:closeFloatLog()

import('com.nx.assist.lua.LuaEngine')

-- ============================================
-- 直接调用方式（不推荐，但可以了解底层实现）

    print('Failed to load APK')
    return
end

    print('Failed to load Class')
    return

local screenHeight = scOps.getScreenHeight()

print('屏幕高度: ' .. tostring(screenHeight))

-- ============================================
-- 实际应用示例：根据屏幕尺寸调整操作
-- ============================================
print('\n========== 实际应用示例 ==========')


if screenWidth and screenHeight then
    print('当前屏幕尺寸: ' .. screenWidth .. ' x ' .. screenHeight)

-- 根据屏幕尺寸判断设备类型
if screenWidth > screenHeight then
    print('当前为横屏模式')
else
    print('当前为竖屏模式')
end

-- 根据屏幕尺寸计算合适的坐标（例如：屏幕中心点）
local centerX = math.floor(screenWidth / 2)
local centerY = math.floor(screenHeight / 2)
print('屏幕中心点: (' .. centerX .. ', ' .. centerY .. ')')

    -- 根据屏幕尺寸计算合适的区域（例如：屏幕右下角 1/4 区域）
    local rightBottomX = math.floor(screenWidth * 3 / 4)
    local rightBottomY = math.floor(screenHeight * 3 / 4)
    print('右下角区域起始点: (' .. rightBottomX .. ', ' .. rightBottomY .. ')')
else
    print('无法获取屏幕尺寸，跳过实际应用示例')
end

print('\n示例执行完成！')

亮度控制 ​

lua
import('com.nx.assist.lua.LuaEngine')
local loader = LuaEngine.loadApk('xfxPlugin-release.apk'); assert(loader)
local context = LuaEngine.getContext()
local UtilsMain = loader.loadClass('com.xfx.plugin.UtilsMain'); assert(UtilsMain)
-- UtilsMain.init(context)

local scOps = UtilsMain.screenOps()
scOps.setContext(context)

print("=== 亮度信息 ===")
local isAuto = scOps.isAutoBrightnessEnabled()
print('自动亮度是否开启: ' .. tostring(isAuto))

if isAuto then
    print('当前使用自动亮度模式，系统会根据环境光自动调整亮度')
else
    print('当前使用手动亮度模式，可以手动设置亮度值')
end

local currentBrightness = scOps.getBrightness()
print('当前系统亮度: ' .. tostring(currentBrightness) .. ' (范围: 0-255)')

-- 计算亮度百分比
local brightnessPercent = math.floor((currentBrightness / 255) * 100)
print('当前亮度百分比: ' .. tostring(brightnessPercent) .. '%')

-- 设置为中等亮度（128，约 50%）
print('准备设置亮度为 128 (50%)...')
local success = pcall(function()
    local result = scOps.setBrightness(128)
    if result then
        print('设置成功，当前亮度: ' .. tostring(scOps.getBrightness()))
    else
        print('设置失败，可能需要相应权限')
    end
end)

if not success then
    print('设置亮度时发生错误，可能需要 ROOT 权限或 WRITE_SETTINGS 权限')
end

print("设置后亮度: " .. tostring(scOps.getBrightness()))

结合 Toast 显示亮度信息 ​

设置上下文的4种方式：

1、分别给每个类设置上下文(下面的例子中已经演示了)。有的类模块是不依赖上下文环境的。 2、UtilsMain.init(LuaEngine.getContext())：设置全局上下文 3、xfxModule:new的时候传入上下文参数来设置全局上下文：xfxModule:new({apkName = 'xfxPlugin-release.apk'， context = LuaEngine.getContext()}) 4、XFX:setContext(LuaEngine.getContext())：设置全局上下文

lua
import('com.nx.assist.lua.LuaEngine')

local xfxModule = require('lib/XfxPlugin')
local XFX = xfxModule:new({apkName = 'xfxPlugin-release.apk'})
XFX:setFloatContext(LuaEngine.getContext())   -- 设置悬浮窗上下文

local scOps = XFX:getOps('screenOps')
if not scOps then
    print('无法获取 scOps 对象')
    return
end

scOps.setContext(LuaEngine.getContext())

-- 获取亮度信息
local brightness = scOps.getBrightness()
local percent = math.floor((brightness / 255) * 100)
local isAuto = scOps.isAutoBrightnessEnabled()

-- 显示 Toast 提示
local message = string.format('当前亮度: %d%%', percent)
if isAuto then
    message = message .. ' (自动)'
end

XFX:showToast(message, 2000)
print(message)

-- 如果亮度过低，给出提示
if brightness < 50 then
    XFX:showToast('亮度较低，建议提高亮度', 3000)
end

sleep(5000)

综合示例：全屏截图 ​

全屏截图依赖 root，使用前建议先用 DeviceOps 的 root 相关接口判断系统状态，再选择合适的截图方式。

下面给出完整流程示例。

示例 1：状态检测 + 自动选择截图方式 ​

lua
import('com.nx.assist.lua.LuaEngine')

local loader = LuaEngine.loadApk('xfxPlugin-release.apk'); assert(loader)
local context = LuaEngine.getContext()
local UtilsMain = loader.loadClass('com.xfx.plugin.UtilsMain'); assert(UtilsMain)
UtilsMain.init(context)

local scOps = UtilsMain.screenOps()
local devOps = UtilsMain.deviceOps()

-- ============================================
-- 1. 用 DeviceOps 判断系统状态（root / adb）
-- ============================================
local isRooted = devOps.isDeviceRooted()       -- 是否已 Root
local isAdb = devOps.isAdbEnabled()            -- ADB 是否开启
local scrcpyReady = scOps.isScrcpyReady()      -- 内置 scrcpy-server.jar 是否可读

print('是否 Root: ' .. tostring(isRooted))
print('ADB 是否开启: ' .. tostring(isAdb))
print('scrcpy 就绪: ' .. tostring(scrcpyReady))

-- 非 root 且无 adb 时，全屏截图基本不可用，提前退出
if not isRooted and not isAdb then
    print('当前环境未 Root 且未开启 ADB，全屏截图不可用')
    return
end

-- ============================================
-- 2. 选择截图方式
--    - 优先 snapShot（scrcpy 失败自动回退 screencap）
--    - 已 Root 时可用 snapShotByShell / snapShotByScrcpy
-- ============================================
local bmp
if scrcpyReady then
    -- 优先使用 scrcpy 特权截图（需 root）
    bmp = scOps.snapShotByScrcpy(0, 0, 0, 0)
    if bmp ~= nil then
        print('scrcpy 截图成功')
    end
end

if bmp == nil then
    -- 回退：root screencap
    bmp = scOps.snapShotByShell(0, 0, 0, 0, isRooted)
    if bmp ~= nil then
        print('shell screencap 截图成功')
    end
end

if bmp == nil then
    -- 最后兜底：snapShot 内部会自动优先 scrcpy，失败回退 screencap
    bmp = scOps.snapShot(0, 0, 0, 0)
end

-- ============================================
-- 3. 结果处理
-- ============================================
if bmp ~= nil then
    print('截图尺寸 w=' .. bmp.getWidth() .. ' h=' .. bmp.getHeight())
    -- 用完后按需 bmp.recycle()
else
    print('截图失败，请确认已 Root / ADB 可用，且 scrcpy-server 正常')
end

示例 2：封装为可复用函数（带状态判断） ​

lua
import('com.nx.assist.lua.LuaEngine')

local xfxModule = require('lib/XfxPlugin')
local XFX = xfxModule:new({
    apkName = 'xfxPlugin-release.apk',
    context = LuaEngine.getContext(),
})

local scOps = XFX:getOps('screenOps')
local devOps = XFX:getOps('deviceOps')

-- 全屏截图（自动判断环境，返回 Bitmap 或 nil）
local function fullScreenShot()
    if not scOps or not devOps then
        print('ops 初始化失败')
        return nil
    end

    -- 状态判断
    local isRooted = devOps.isDeviceRooted()
    local scrcpyReady = scOps.isScrcpyReady()

    if not isRooted and not scrcpyReady then
        print('环境不支持全屏截图：未 Root 且 scrcpy 不可用')
        return nil
    end

    -- 优先 scrcpy，失败自动回退 screencap
    local bmp = scOps.snapShot(0, 0, 0, 0)
    return bmp
end

local bmp = fullScreenShot()
if bmp ~= nil then
    XFX:showToast('截图成功 ' .. bmp.getWidth() .. 'x' .. bmp.getHeight(), 1500)
    -- bmp.recycle()
else
    XFX:showToast('截图失败，请确认已 Root', 1500)
end

sleep(2000)
